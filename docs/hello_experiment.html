    html: (root, el) =>
      $('#experiment').empty()
      $(root).html(el)

      container = @$('[data-elem="sketch-on-me"]')[0]
      @ctx = Sketch.create
        container: container
        autoclear: false
        autostart: false

      @ctx.width = 800
      @ctx.height = 250
      @ctx.draw = =>
        @ctx.beginPath()
        @ctx.arc( random( @ctx.width ), random( @ctx.height ), 3, 0, TWO_PI )
        @ctx.fillStyle = '#776d6b'
        @ctx.fill()

      @ctx.start()

    close: (e) ->
      @remove()
      @ctx.stop()
      @ctx.clear()
      Backbone.history.navigate('/',true)
